This test consists of four sections with five questions each.
Each question is multiple-choice with five possible answers of which exactly one is correct.
Each question is worth $1$ mark; an incorrect answer or a blank response will be awarded $0$ marks.

\section{Algebra}
\begin{enumerate}

\item % Ralph, correct answer: \sqrt{6}
Let $x$, $y$ and $z$ be positive real numbers that satisfy $xy = 1$, $yz = 2$, $zx = 3$.
What the value of $xyz$? 
\answers{$\sqrt{2}$}{$2$}{$\sqrt{6}$}{$6$}{Impossible}

\item % Ralph, correct answer: \frac{9}{10}
Given that $x$ satisfies $x + \frac{1}{x} = 10$, what is the value of $\frac{x^2-x+1}{x^2 +1}$?
\answers{$\frac{1}{100}$}{$\frac{9}{10}$}{$\frac{91}{100}$}{$\frac{99}{100}$}{$1$}

\item % Ralph, correct answer: 7
How many roots (not necessarily unique) does the polynomial $P(x) = x^7-5x^3 + x + 1$ have in the complex numbers?
\answers{Unknown}{$1$}{$3$}{$5$}{$7$}

\item % Ralph, correct answer: 2021^\frac{2022}{2021}
Given a sequence of positive real numbers $a_1$, $a_2$, ..., $a_{2021}$ satisfying $a_1a_2 \cdots a_{2021} = 2021$, what is the minimum of $\sum_{i = 1}^{2021} a_i$?
\answers{$2021^\frac{2022}{2021}$}{$2021$}{$1$}{$2021 + \frac{1}{2021}$}{$0$}

\item % Ralph, correct answer: 3 (-4, 0, 4)
Consider a function $f: \mathbb{R} \rightarrow \mathbb{R}$ that satisfies $f(x + y)^2 = f(x^2) + f(y^2)$ for all $x$, $y \in \mathbb{R}$.
How many possible values are there for the expression $f(x) + f(y)$?
\answers{$0$}{$1$}{$2$}{$3$}{$4$}

\end{enumerate}



\section{Combinatorics}
\begin{enumerate}

\item % Ralph, correct answer: 2
Let $A = \{2, 3, 5, 7\}$ and $B = \{3, 4, 5, 6 \}$; what is $|A \cap B|$?
\answers{$0$}{$2$}{$4$}{$6$}{$8$}

\item % Ralph, correct answer: 2^{32}
Given a set of all of the $1 \times 1$ squares on an $8 \times 8$ chessboard (each square is assumed to be unique), how many subsets of this set do not contain any white squares?
\answers{$0$}{$32$}{$2^{32}$}{$32!$}{$2^{64}$}

\item % Ralph, correct answer: \mathbb{R}
SAMF offers Liam five payment options (A-E) for his work at the camp.
These will all be in the form of a bag full of R1 coins.
Each bag is labelled with a set indicating that Liam can construct a bijection between the coins in the bag and that set.
Which bag is the biggest? 
\answers{$\mathbb{N}$}{$\mathbb{Z}$}{$\mathbb{Q}$}{$\emptyset$}{$\mathbb{R}$} 

\item % Aaron, correct answer: 7 
What is the smallest natural number $n$ such that any set of $n$ integers will always contain two numbers whose sum or difference is a multiple of $10$?
\answers{$5$}{$6$}{$7$}{$10$}{$11$}

\item % Aaron, correct answer: 57 
The numbers $1$, $2$, ..., $2021$ are written on a blackboard.
Every minute, Aaron erases two numbers $a$ and $b$ and replaces them with the single number $|a - b|$.
Which of the following could be the number he has on the board after $2020$ minutes?
\answers{$156$}{$57$}{$48$}{$1024$}{$2020$}


\end{enumerate}



\section{Geometry}
\begin{enumerate}

\item % Ralph, correct answer 90^\circ 
Let $ABC$ and $ABD$ be two isosceles triangle with $C$, $B$, and $D$ collinear and $C \ne D$.
What is the value of $\angle CAD$?
\answers{$72^\circ$}{$135^\circ$}{$90^\circ$}{$108^\circ$}{Impossible}

\item %  Ralph, correct answer \frac{1}{20}
Let $ABCD$ be a square, and let $M$ and $N$ be the midpoints of $BC$ and $CD$ respectively.
Let $P$ be the intersection of $AM$ and $BN$.
What fraction of the square's area is $\triangle BPM$?
\answers{ $\frac{1}{8}$ }{ $\frac{1}{100}$ }{ $\frac{1}{24}$ }{ $\frac{1}{10}$ }{ $\frac{1}{20}$ }

\item % Ralph, correct answer 2
Consider the following five statements:
\begin{itemize}
  \item In non-degenerate triangle $ABC$, it is guaranteed that $|AB| > |BC| + |CA|$.
  \item The incentre, centroid, and circumcentre of a triangle are collinear.
  \item It is possible to divide an arbitrary angle into three equal parts using only a ruler and compass.
  \item In quadrilateral $ABCD$, it is always true that $|AB| \cdot |CD| + |AD| \cdot |BC| \ge |AC| \cdot |BD|$.
  \item It is possible to inscribe a circle inside any non-degenerate triangle, and the centre of this circle is the intersection of perpendicular bisectors of the sides of the triangle.
\end{itemize}
How many of these are correct?
\answers{$1$}{$2$}{$3$}{$4$}{$5$}

\item % Ralph, correct answer 15
In triangle $ABC$, let $H$ be the orthocentre.
Let $A'$, $B'$, and $C'$ be the reflections of $H$ over lines $BC$, $CA$, and $AB$ respectively.
How many cyclic quadrilaterals are there that use only the points $A$, $B$, $C$, $H$, $A'$, $B'$, $C'$?
\answers{$15$}{$35$}{$0$}{$4$}{$7$}

\item % Ralph, correct answer \frac{11}{10}
Consider $\triangle ABC$ with $|AB| = 8$, $|BC| = 9$, and $|CA| = 6$.
Let $P$ be a point inside $\triangle ABC$ and let $D$, $E$, and $F$ be the intersections of the extensions of $AP$, $BP$, and $CP$ with $BC$, $CA$, and $AB$ respectively.
If $|AE| = 2$, and $|AF| = 3$, what is $\frac{|AP|}{|PD|}$?
\answers{ $\frac{11}{10}$ }{ $\frac{3}{2}$ }{ $\frac{9}{8}$ }{ $\frac{14}{9}$ }{ $\frac{5}{4}$ }

\end{enumerate}



\section{Number Theory}
\begin{enumerate}

\item % Ralph, correct answer 119
What is the largest positive integer that cannot be written as the sum of non-negative multiples of $11$ and $13$?
\answers{$119$}{$93$}{$73$}{$2022$}{$128$}

\item % Dylan, correct answer 720
How many natural numbers below $1001$ are relatively prime to $1001$?
\answers{$168$}{$720$}{$852$}{$900$}{$1000$}

\item % Ralph, correct answer 1
Consider the following four statements:
\begin{itemize}
  \item Every positive integer can be written as the sum of two squares.
  \item If $a \mid b$ and $a \mid c$, then $b \mid c$.
  \item Given $a$, $b \in \mathbb{N}$, we have that $\lcm(a, b) = \frac{ab}{\gcd(a, b)}$.
  \item All prime numbers are odd.
\end{itemize}
How many of these are correct?
\answers{0}{1}{2}{3}{4}

\item % Ralph, correct answer 8585
$k \in \mathbb{N}$ is chosen such that $k^2 + 2016k + 2015$ is a perfect square.
Which of the following is a possible value for $k$?
\answers{$1544$}{$3471$}{$6190$}{$8585$}{$9381$}

\item % Ralph, correct answer 43
What are the last two digits of $7^{7^{7^7}}$?
\answers{$01$}{$07$}{$49$}{$43$}{$85$}

\end{enumerate}
