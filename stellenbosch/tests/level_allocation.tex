\section{Algebra}
\begin{enumerate}[1.]

\item % Ralph, correct answer: \sqrt{6}
Let $x$, $y$ and $z$ satisfy $xy = 1$, $yz = 2$, $zx = 3$.
What the value of $xyz$? 
\answers{$\sqrt{2}$}{$2$}{$\sqrt{6}$}{$6$}{Impossible}


\item % Ralph, correct answer: \frac{9}{10}
Given that $x$ satisfies $x + \frac{1}{x} = 10$, what is the value of $\frac{x^2-x+1}{x^2 +1}$?
\answers{$\frac{1}{100}$}{$\frac{9}{10}$}{$\frac{91}{100}$}{$\frac{99}{100}$}{$1$}


\item % Ralph, correct answer: 7
How many roots (not necessarily unique) does the polynomial $P(x) = x^7-5x^3 + x + 1$ have?
\answers{Unknown}{$1$}{$3$}{$5$}{$7$}


\item % Ralph, correct answer: 2021^\frac{2022}{2021}
Given a sequence $a_1$, $a_2$, $\cdots$, $a_{2021}$ satisfying $a_1a_2 \cdots a_{2021} = 2021$, and $a_{i} \in \mathbb{R}^+ \; \forall i \in \{1, 2, \cdots, 2021\}$, what is the minimum of $\sum_{i = 1}^{2021} a_i$?
\answers{$0$}{$1$}{$2021$}{$2021 + \frac{1}{2021}$}{$2021^\frac{2022}{2021}$}


\item % Ralph, correct answer: 4 (-4, -2, 0, 2, 4)
It is known that a function $f: \mathbb{R} \rightarrow \mathbb{R}$ satisfies $f(x + y)^2 = f(x^2) + f(y^2)$ for all $x$, $y \in \mathbb{R}$.
How many possible values are there for the expression $f(x) + f(y)$?
\answers{0}{1}{2}{3}{4}

\end{enumerate}
